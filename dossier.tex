\begin{algorithm}[H]
  \KwData{Un ensemble de sommets V représentant les protéines et un ensemble de sous-complexes C}
  \KwResult{Un graphe représentant la solution optimale au problème IC sous forme de liste d'adjascence}
  $graph \gets list()$\;
  \ForEach{subcomplex $C_i \in C$ }{
    \ForEach{protein $p_{i_j} \in C_i$}{
      \If{$j < size(C_i) - 1$}{

    }
